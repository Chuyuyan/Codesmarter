// Complete diagnostic script
console.log('=== TAB DIAGNOSTIC ===');

// Check if tabs exist in DOM
const tabs = document.querySelector('.index-tabs');
const uploadTab = document.querySelector('#upload-tab');
const gitTab = document.querySelector('#git-tab');
const pathTab = document.querySelector('#path-tab');

console.log('\n1. Element Check:');
console.log('  Tabs container:', tabs ? '‚úÖ EXISTS' : '‚ùå NOT FOUND');
console.log('  Upload tab:', uploadTab ? '‚úÖ EXISTS' : '‚ùå NOT FOUND');
console.log('  Git tab:', gitTab ? '‚úÖ EXISTS' : '‚ùå NOT FOUND');
console.log('  Path tab:', pathTab ? '‚úÖ EXISTS' : '‚ùå NOT FOUND');

if (tabs) {
    console.log('\n2. Tabs Container Info:');
    const style = window.getComputedStyle(tabs);
    console.log('  Display:', style.display);
    console.log('  Visibility:', style.visibility);
    console.log('  Opacity:', style.opacity);
    console.log('  Width:', style.width);
    console.log('  Height:', style.height);
    console.log('  HTML:', tabs.outerHTML.substring(0, 200));
    
    console.log('\n3. Tab Buttons:');
    const buttons = tabs.querySelectorAll('.tab-btn');
    console.log('  Button count:', buttons.length);
    buttons.forEach((btn, i) => {
        console.log(`  Button ${i + 1}:`, btn.textContent.trim());
    });
} else {
    console.log('\n‚ùå PROBLEM: Tabs container not found in DOM!');
    console.log('This means the HTML is old version.');
    
    // Check what's actually in the index-section
    const indexSection = document.querySelector('#index-section');
    if (indexSection) {
        console.log('\nCurrent index-section content:');
        console.log(indexSection.innerHTML.substring(0, 500));
    }
}

// Check page HTML source
console.log('\n4. Page Source Check:');
const pageHTML = document.documentElement.outerHTML;
if (pageHTML.includes('index-tabs')) {
    console.log('  ‚úÖ HTML source contains "index-tabs"');
} else {
    console.log('  ‚ùå HTML source does NOT contain "index-tabs"');
    console.log('  This confirms old HTML version is loaded.');
}

// Check if cache is the issue
console.log('\n5. Cache Check:');
console.log('  Current URL:', window.location.href);
console.log('  Page title:', document.title);
